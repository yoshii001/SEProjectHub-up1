## 1. Firebase Configuration Verification

### Step 1.1: Check Firebase Project Configuration

**Verify your Firebase configuration object:**

```javascript
// src/lib/firebase.ts
const firebaseConfig = {
  apiKey: "your-api-key",
  authDomain: "your-project.firebaseapp.com",
  databaseURL: "https://your-project-default-rtdb.firebaseio.com", // ‚úÖ CRITICAL: Must include databaseURL
  projectId: "your-project-id",
  storageBucket: "your-project.firebasestorage.app",
  messagingSenderId: "123456789",
  appId: "your-app-id"
};
```

**Common Issues:**
- ‚ùå Missing `databaseURL` field
- ‚ùå Incorrect region in database URL
- ‚ùå Wrong project ID

**Verification Steps:**
1. Go to Firebase Console ‚Üí Project Settings ‚Üí General
2. Copy the exact configuration from "Your apps" section
3. Ensure `databaseURL` matches your database region

### Step 1.2: Verify Database Import

```javascript

// ‚ùå Wrong import (this is for Firestore)

```

## 2. Database Rules Setup

### Step 2.1: Basic Security Rules

Go to Firebase Console ‚Üí Realtime Database ‚Üí Rules:

```json
{
  "rules": {
    ".read": "auth != null",
    ".write": "auth != null",
    "clients": {
      "$clientId": {
        ".read": "data.child('userId').val() === auth.uid",
        ".write": "data.child('userId').val() === auth.uid || (!data.exists() && newData.child('userId').val() === auth.uid)"
      }
    },
    "projects": {
      "$projectId": {
        ".read": "data.child('userId').val() === auth.uid",
        ".write": "data.child('userId').val() === auth.uid || (!data.exists() && newData.child('userId').val() === auth.uid)"
      }
    },
    "meetings": {
      "$meetingId": {
        ".read": "data.child('userId').val() === auth.uid",
        ".write": "data.child('userId').val() === auth.uid || (!data.exists() && newData.child('userId').val() === auth.uid)"
      }
    }
  }
}
```

### Error 4.1: "Permission denied"

**Error Message:**
```
FIREBASE WARNING: Exception was thrown by user callback. Error: Permission denied
```

**Solutions:**
1. Check authentication state
2. Verify database rules
3. Ensure user is logged in

### Error 4.3: "Invalid configuration"

**Error Message:**
```
Error: Invalid Firebase configuration
```

**Solutions:**
1. Verify all required config fields
2. Check for typos in configuration
3. Ensure correct project selected

```javascript
// Validate Firebase configuration

  const requiredFields = [
    'apiKey',
    'authDomain',
    'databaseURL',
```

### Example 5.3: Custom Hook with Error Handling

```javascript
// src/hooks/useFirebaseConnection.ts
import { useState, useEffect } from 'react';
import { ref, onValue, off } from 'firebase/database';
import { database } from '../lib/firebase';

export const useFirebaseConnection = () => {
  const [isConnected, setIsConnected] = useState(false);
  const [connectionError, setConnectionError] = useState<string | null>(null);

  useEffect(() => {
    const connectedRef = ref(database, '.info/connected');
    
    const unsubscribe = onValue(connectedRef, 
      (snapshot) => {
        const connected = snapshot.val();
        setIsConnected(connected);
        setConnectionError(null);
        
        if (connected) {
          console.log('‚úÖ Firebase connected');
        } else {
          console.log('‚ùå Firebase disconnected');
        }
      },
      (error) => {
        console.error('‚ùå Connection error:', error);
        setConnectionError(error.message);
        setIsConnected(false);
      }
    );

    return () => off(connectedRef, 'value', unsubscribe);
  }, []);

  return { isConnected, connectionError };
};
```

## 6. Authentication State Validation

### Step 6.1: Check Authentication State

```javascript
// src/hooks/useAuthState.ts
import { useState, useEffect } from 'react';
import { onAuthStateChanged, User } from 'firebase/auth';
import { auth } from '../lib/firebase';

export const useAuthState = () => {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    console.log('üîÑ Setting up auth state listener');
    
    const unsubscribe = onAuthStateChanged(
      auth,
      (user) => {
        console.log('üë§ Auth state changed:', user ? 'Logged in' : 'Logged out');
        if (user) {
          console.log('üìß User email:', user.email);
          console.log('üÜî User ID:', user.uid);
        }
        setUser(user);
        setLoading(false);
        setError(null);
      },
      (error) => {
        console.error('‚ùå Auth state error:', error);
        setError(error.message);
        setLoading(false);
      }
    );

    return () => {
      console.log('üîÑ Cleaning up auth state listener');
      unsubscribe();
    };
  }, []);

  return { user, loading, error };
};
```
